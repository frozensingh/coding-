1st => BUBBLE SHORT


#include<stdio.h>

void Swap(int *x,int *y){
    int temp=*x;
    *x=*y;
    *y=temp;
    
}

void Bubble(int A[],int n){
    int flag,i,j;
    for(i=0;i<(n-1);i++){
        flag=0;
        for(j=0;j<n-1-i;j++){
            if(A[j]>A[j+1]){
                Swap(&A[j],&A[j+1]);
                flag++;  }}
                if(flag==0)
                    break;
    } }    
    
int main(void)
{
    int i,A[]={2,4,3,5,9,6,7};
    Bubble(A,7);
    for(i=0;i<7;i++){
        printf(" %d",A[i]);
    }
    return 0;
}

2nd => insertion SHORT

#include<stdio.h>

void insertion(int A[],int n){
    int x,i,j;
    for(i=0;i<n;i++) {
        j=i-1;
        x=A[i];
        while(j>-1&&A[j]>x){
            A[j+1]=A[j];
            j--;
        }
         A[j+1]=x;   
    } }    
    
int main(void)
{
    int i,A[]={2,4,3,5,9,6,7};
    insertion(A,7);
    for(i=0;i<7;i++){
        printf(" %d",A[i]);
    }
    return 0;
}

#include<stdio.h>
#include<stdlib.h>    

int findMax(int A[],int n){
    int max ;
    int i;
    for(i=0;i<n;i++){
        if(A[i]>max)
            max=A[i];
    }
    return max;
}

void CountSort(int A[],int n){
    int i,j,max,*c;
    max=findMax(A,n);
    c=(int*)malloc(sizeof(int)*(max+1));
    for(i=0;i<max+1;i++)
        {
            c[i]=0;
        }
     for(i=0;i<n;i++)
    {
    c[A[i]]++;
    }
    i=0;j=0;
    while(j<max+1){
        if(c[j]>0){
            A[i++]=j;
            c[j]--;
        }
        else
            j++;
    }}
    
int main(void)
{
    int i,A[]={2,4,3,5,9,6,7};
    CountSort(A,7);
    for(i=0;i<7;i++){
        printf(" %d",A[i]);
    }
    return 0;
}
